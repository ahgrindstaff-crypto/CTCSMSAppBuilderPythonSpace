import tkinter as tk from tkinter import messagebox, scrolledtext import re class SpaceCommander: def __init__(self, root): self.root = root self.root.title("Python Space Commander - Desktop Mission Control") self.root.geometry("1000x700") self.root.configure(bg="#0b0d17") self.current_level = 0 self.levels = [ {"task": "Create a variable 'ship_name' and set it to 'Apollo'.", "check": r"ship_name\s*=\s*['\"]Apollo['\"]", "success": "Ship name registered: Apollo"}, {"task": "Set the destination: destination = 'Mars'", "check": r"destination\s*=\s*['\"]Mars['\"]", "success": "Trajectory set for Mars."}, {"task": "Print the ship name: print(ship_name)", "check": r"print\s*\(\s*ship_name\s*\)", "success": "Apollo"}, {"task": "Set fuel level to 100: fuel = 100", "check": r"fuel\s*=\s*100", "success": "Fuel tanks at 100%."}, {"task": "Print: print('Engines Warming Up...')", "check": r"print\s*\(\s*['\"]Engines Warming Up...['\"]", "success": "Engines Warming Up..."}, {"task": "Update distance: distance = distance + 500", "check": r"distance\s*\+\s*500", "success": "Distance updated: 500ly"} # Add more levels here following the same pattern ] self.setup_ui() def setup_ui(self): # Header self.header = tk.Frame(self.root, bg="#1a237e", height=60) self.header.pack(fill="x") self.lbl_title = tk.Label(self.header, text="Mission Control: Level 1", fg="white", bg="#1a237e", font=("Courier", 18, "bold")) self.lbl_title.pack(side="left", padx=20, pady=10) self.lbl_progress = tk.Label(self.header, text=f"Level 1 of {len(self.levels)}", fg="white", bg="#1a237e", font=("Courier", 12)) self.lbl_progress.pack(side="right", padx=20) # Main Container self.main_frame = tk.Frame(self.root, bg="#0b0d17") self.main_frame.pack(expand=True, fill="both", padx=20, pady=20) # Left Side (Editor) self.left_frame = tk.Frame(self.main_frame, bg="#0b0d17") self.left_frame.pack(side="left", expand=True, fill="both", padx=(0, 10)) self.hint_box = tk.Label(self.left_frame, text=self.levels[0]["task"], bg="#263238", fg="#eceff1", font=("Courier", 12), wraplength=400, justify="left", anchor="nw", padx=10, pady=10, bd=2, relief="flat") self.hint_box.pack(fill="x", pady=(0, 10)) self.code_editor = scrolledtext.ScrolledText(self.left_frame, bg="#1c1c1c", fg="#00ff41", insertbackground="white", font=("Consolas", 14), undo=True) self.code_editor.pack(expand=True, fill="both") # Buttons self.btn_frame = tk.Frame(self.left_frame, bg="#0b0d17") self.btn_frame.pack(fill="x", pady=10) self.run_btn = tk.Button(self.btn_frame, text="Initiate Launch (Run)", bg="#1b5e20", fg="white", font=("Courier", 12, "bold"), command=self.process_code, cursor="hand2") self.run_btn.pack(side="left", expand=True, fill="x", marginRight=5) self.next_btn = tk.Button(self.btn_frame, text="Next Level →", bg="#0277bd", fg="white", font=("Courier", 12, "bold"), command=self.next_level, cursor="hand2") # Hidden initially # Right Side (Console) self.right_frame = tk.Frame(self.main_frame, bg="#0b0d17") self.right_frame.pack(side="right", expand=True, fill="both", padx=(10, 0)) tk.Label(self.right_frame, text="Terminal Output:", bg="#0b0d17", fg="#cfd8dc", font=("Courier", 12)).pack(anchor="w") self.terminal = tk.Text(self.right_frame, bg="black", fg="#00ff41", font=("Consolas", 12), state="disabled", padx=10, pady=10) self.terminal.pack(expand=True, fill="both") self.log_terminal("System Ready... Waiting for input.") def log_terminal(self, message, color="#00ff41"): self.terminal.config(state="normal") self.terminal.insert("end", f"> {message}\n") self.terminal.config(state="disabled") self.terminal.see("end") def process_code(self): user_code = self.code_editor.get("1.0", "end-1.c") current_level_data = self.levels[self.current_level] # Use Regex to validate input if re.search(current_level_data["check"], user_code, re.IGNORECASE): self.log_terminal(current_level_data["success"]) self.log_terminal("✅ [MISSION CLEAR] Requirement Met.") self.next_btn.pack(side="left", expand=True, fill="x", padx=5) else: self.log_terminal("❌ PYTHON ERROR: Syntax or Logic mismatch.", color="red") def next_level(self): self.current_level += 1 if self.current_level < len(self.levels): # Update UI for next level self.lbl_title.config(text=f"Mission Control: Level {self.current_level + 1}") self.lbl_progress.config(text=f"Level {self.current_level + 1} of {len(self.levels)}") self.hint_box.config(text=self.levels[self.current_level]["task"]) self.code_editor.delete("1.0", "end") self.terminal.config(state="normal") self.terminal.delete("1.0", "end") self.terminal.config(state="disabled") self.log_terminal(f"Initializing Level {self.current_level + 1}...") self.next_btn.pack_forget() else: messagebox.showinfo("Galaxy Edge Reached", "Commander, you've completed all missions! The galaxy is safe.") self.root.quit() if __name__ == "__main__": root = tk.Tk() app = SpaceCommander(root) root.mainloop()
